    Basic Syntax

    Scala

    // def to define functions/methods
def isTrue(b: Boolean) = b // implicit return type

// [] instead of <>
val emptyList = List[String]()

// Scala's == != Java's ==
val p1 = new Point(1, 2)
val p2 = new Point(1, 2)
p1 == p2 // true
p1 eq p2 // false

// no handled exceptions
def mayFail = throw new Exception

    Groovy

    // implict public
boolean isTrue(boolean value) { value } // implicit return statement

// implicit return type (Object)
def isTrue(boolean value) { value }

// implicit parameter types (Object)
Object isTrue(value) { value }

// generics like Java (but optional)
ArrayList myList = new ArrayList<String>()

// == refers to equals()
// obj.is(obj2) refers to java's ==
Point p1 - new Point(1,2)
Point p2 = new Point(1,2)
p1 == p2        // true
p1.is(p2)    // false

// All exceptions are treated as RuntimeException
def mayFail() {
    throw new Exception()
}

    Classes and Stuff

    Scala

    // mixins, like interfaces, but with implementation
trait Flyer {
    def fly = "I'm flying"
}
trait Quacker {
    def quack = "Quack!"
}

// yeah, you know...
class Duck extends Flyer with Quacker
val d = new Duck
d.fly
d.quack

// single instance of a type (singleton)
object A
var a1 = A
var a2 = A
a1 eq a2 // true

// a class with lots of code generated by the compiler
// equals, toString, hashCode, apply, unapply
case class Point(x: Int, y: Int)

    Groovy

    class Quacker {
    def quack() { 'Quack!' }
}

class Flyer {
    def fly() { 'Fly' }
}

@Mixin([Quacker, Flyer])
class Duck {}

def duck = new Duck()
duck.fly()
duck.quack()

@Singleton
class Plateau {
    @Lazy
    def expensiveResource = {
        println 'Expensive resource created'
    }()
}

//new Plateau()    // fails!
def plateau1 = Plateau.instance
def plateau2 = Plateau.instance
assert plateau1.is(plateau2)

    Scala

    Getters and Setters

    Scala

    case class Number(i: Int)

object NumberOneHolder {
	
    private var _n: Number = _

    def n = _n
    def n_=(nn: Number): Unit = nn match {
        case Number(1) => _n = nn
        case _ => error("This Number != 1")
    }
}

NumberOneHolder.n = Number(1)
NumberOneHolder.n = Number(2) // error: This Number != 1

    Pattern Matching

    Scala

    case class Witch(name: String)
val KnightWhoSayRE = """^Knights Who Say (.*)""".r
def findFromTheMovie(a: Any) = a match {
    case "Bedemir" | "Galahad" | "Lancelot" | "Robin" => 
        "Found a Knight"
    case Witch(name) => 
        "Found a Witch named " + name
    case KnightWhoSayRE(what) => 
        "Found the Knights Who Say " + what
    case i: Int if i > 0 => 
        "Found the speed of a swallow"
    case <montyPython>{sirs @ _*}</montyPython> => 
        "Found a montyPython XML"
    case _ => 
        "Found nothing"
}

findFromTheMovie("Robin")
findFromTheMovie(Witch("a witch"))

    Groovy

    //pattern on righthand side between single-slashes
assert 'abc' ==~ /abc/
assert 'abc' ==~ /a.c/

//turn on flag halfway thru pattern
assert 'abcDEFG' ==~ /abc(?i)defg/

//turn flag on, then off again
assert 'abCDefg' ==~ /ab(?i)cd(?-i)efg/
